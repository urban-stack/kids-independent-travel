Visualizations

fn_plot_preds_c <- function(my_preds,
                          to_vary) {
  my_preds |>
    filter(vary == to_vary) |>
    select(c(matches(to_vary), starts_with("Prob. "))) |>
    pivot_longer(cols = starts_with("Prob. "),
                 names_to = "Mode",
                 values_to = "Probability",
                 names_prefix = "Prob. ") |>
    mutate(Mode = factor(Mode, 
                         levels = c("alone", 
                                    "mom",
                                    "mom and dad",
                                    "dad",
                                    "non-hh",
                                    "sibling"),
                         labels = c("Alone", 
                                    "With mom",
                                    "With both mom and dad",
                                    "With dad",
                                    "With a non-household member",
                                    "With a sibling"))) |>
    rename(to_vary = !!to_vary) |>
    ggplot() +
    geom_area(aes(x = to_vary,
                  y = Probability,
                  fill = Mode)) +
    scale_fill_brewer(type = "qual") +
    theme_minimal() +
    scale_y_continuous(breaks = c(0, 0.5, 1),
                       labels = c("0", "50%", "100%")) 
}

fn_plot_preds_b <- function(my_preds,
                          to_vary) {
  my_preds |>
    filter(vary == to_vary) |>
    select(c(matches(to_vary), starts_with("Prob. "))) |>
    pivot_longer(cols = starts_with("Prob. "),
                 names_to = "Mode",
                 values_to = "Probability",
                 names_prefix = "Prob. ") |>
    mutate(Mode = factor(Mode, 
                         levels = c("alone", 
                                    "mom",
                                    "mom and dad",
                                    "dad",
                                    "non-hh",
                                    "sibling"),
                         labels = c("Alone", 
                                    "With mom",
                                    "With both mom and dad",
                                    "With dad",
                                    "With a non-household member",
                                    "With a sibling"))) |>
    rename(to_vary = !!to_vary) |>
    ggplot() +
    geom_col(aes(x = as_factor(to_vary),
                  y = Probability,
                  fill = Mode),
             position = "stack") +
    scale_fill_brewer(type = "qual") +
    theme_minimal() +
    scale_y_continuous(breaks = c(0, 0.5, 1),
                       labels = c("0", "50%", "100%")) 
}

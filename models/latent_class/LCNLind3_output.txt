function(apollo_beta, apollo_inputs, functionality = "estimate") {
  
  ### Attach inputs and detach after function exit
  apollo_attach(apollo_beta, apollo_inputs)
  on.exit(apollo_detach(apollo_beta, apollo_inputs))
  
  ### Create list of probabilities P
  P = list()
  
  # create nests
  nests <- list()
  
  # first nest: group by independence
  nests[["a"]]$nlNests <- list(root = 1, Alone = lambda_alone, Parent = lambda_parent, Others = lambda_others)
  nests[["b"]]$nlNests <- list(root = 1, Walk  = lambda_walk,  Bike = lambda_bike,  Drive = lambda_drive)
  
  ### Specify tree structure for NL model
  nests[["a"]]$nlStructure = list()
  nests[["a"]]$nlStructure[["root"]]    = c("Alone", "Parent", "Others")
  nests[["a"]]$nlStructure[["Alone"]]   = c("walk_alone","bike_alone")
  nests[["a"]]$nlStructure[["Parent"]]  = c("walk_parent","bike_parent", "drive_parent")
  nests[["a"]]$nlStructure[["Others"]]  = c("walk_others","bike_others", "drive_others")
  
  nests[["b"]]$nlStructure = list()
  nests[["b"]]$nlStructure[["root"]]  = c("Walk", "Drive", "Bike")
  nests[["b"]]$nlStructure[["Walk"]]  = c("walk_alone","walk_others", "walk_parent")
  nests[["b"]]$nlStructure[["Drive"]] = c("drive_others", "drive_parent")
  nests[["b"]]$nlStructure[["Bike"]]  = c("bike_alone","bike_others", "bike_parent")

  ## loop over classes
  for(s in 1:2){
    
    ### compute class-specific utilities
    V = list()
    V[["drive_parent"]]  = asc_dp[[s]] + beta_nadults_do[[s]] * n_adults
    V[["drive_others"]]  = asc_do[[s]] + beta_nadults_dp[[s]] * n_adults
    V[["walk_alone"]]    = asc_wa[[s]] + beta_nadults_wa[[s]] * n_adults
    V[["walk_parent"]]   = asc_wp[[s]] + beta_nadults_wp[[s]] * n_adults
    V[["walk_others"]]   = asc_wo[[s]] + beta_nadults_wo[[s]] * n_adults
    V[["bike_alone"]]    = asc_ba[[s]] + beta_nadults_ba[[s]] * n_adults
    V[["bike_parent"]]   = asc_bp[[s]] + beta_nadults_bp[[s]] * n_adults
    V[["bike_others"]]   = asc_bo[[s]] + beta_nadults_bo[[s]] * n_adults
    
    ### Define settings for NL model
    nl_settings = list(
      alternatives = c(
        walk_alone = "walk_alone", walk_others = "walk_others", walk_parent = "walk_parent",
        bike_alone = "bike_alone", bike_others = "bike_others", bike_parent = "bike_parent",
        drive_others = "drive_others", drive_parent = "drive_parent"
      ),
      choiceVar    = choice,
      explanators  = c("female", "n_adults"),
      utilities = V,
      nlStructure = nests[[s]]$nlStructure,
      nlNests = nests[[s]]$nlNests,
      componentName = paste0("Class_", s)
    )
    
    ### Compute within-class choice probabilities using MNL model
    P[[paste0("Class_",s)]] = apollo_nl(nl_settings, functionality)
  }
  
  # compute latent class model probabilities
  lc_settings = list(inClassProb = P, classProb = pi_values)
  P[["model"]] = apollo_lc(lc_settings, apollo_inputs, functionality)
  
  ### Prepare and return outputs of function
  P = apollo_prepareProb(P, apollo_inputs, functionality)
  return(P)
}
